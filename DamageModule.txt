-- Damage Module v2 (Robust Parsing)
local Players = game:GetService("Players")

game.Players.PlayerAdded:Connect(function(player)
	player.Chatted:Connect(function(message)
		local command = string.lower(message)

		-- Safe Parsing: Split string by space
		local parts = string.split(command, " ")

		-- Check if command is "/sethealth"
		if parts[1] == "/sethealth" then
			local healthValue = tonumber(parts[2]) -- Gets the number after space

			if healthValue and player.Character and player.Character:FindFirstChild("Humanoid") then
				-- Update Health
				player.Character.Humanoid.Health = healthValue
				print("[DAMAGE MOD] Health set to: " .. tostring(healthValue))
			else
				warn("[DAMAGE MOD] Failed to set health.")
			end
		end
	end)
end)
